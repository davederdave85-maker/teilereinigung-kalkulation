function exportWord() { const data = collectAngebotData(); function esc(text) { return String(text || '') .replace(/&/g, '&') .replace(//g, '>') .replace(/"/g, '"'); } function eur(value) { return (Number(value) || 0).toLocaleString('de-DE', { minimumFractionDigits: 2, maximumFractionDigits: 2 }) + ' €'; } const empPositions = data.positions.filter(p => !p.isOptional); const totalEmp = empPositions.reduce((sum, p) => sum + (Number(p.offerPrice) || 0), 0); let currentGroup = ''; let rowsHtml = ''; data.positions.forEach((pos, index) => { if (pos.groupTitle !== currentGroup) { currentGroup = pos.groupTitle; rowsHtml += ` ${esc(currentGroup)} `; } rowsHtml += ` ${index + 1}. ${esc(pos.name)} ${pos.isOptional ? eur(pos.offerPrice) : ''} ${!pos.isOptional ? eur(pos.offerPrice) : ''} `; }); const overheadRows = (data.overhead || []).map((oh, i) => ` ${51 + i}. ${esc(oh.name)} ${(Number(oh.price) || 0) > 0 ? eur(oh.price) : 'TBD'} `).join(''); const notesHtml = esc(data.notes || '').replace(/\n/g, '
'); const content = ` Angebot

PREISÜBERSICHT in EURO

Angebot Nr.: ${esc(data.offerNo || '—')}
Datum: ${esc(data.offerDate || '—')}
Kunde: ${esc(data.custName || '—')}
Ansprechpartner: ${esc(data.custContact || '—')}
Baureihe: ${esc(data.seriesLabel || '—')}
Betreff: ${esc(data.offerSubject || '—')}
${rowsHtml}
Position Optional Empfohlen
Anlagenpreis in empfohlener Ausstattung ${eur(totalEmp)}
${(data.overhead && data.overhead.length) ? `
OVERHEAD
${overheadRows}
` : ''}
Weitere Optionen und Varianten auf Anfrage.
${data.notes ? `
Bemerkungen:
${notesHtml}
` : ''} `; const blob = new Blob(['\ufeff', content], { type: 'application/msword' }); const link = document.createElement('a'); link.href = URL.createObjectURL(blob); link.download = `Angebot_${(data.offerNo || 'Export').replace(/[^\w\-]+/g, '_')}.doc`; link.click(); }
